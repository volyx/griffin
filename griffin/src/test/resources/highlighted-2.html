class Solution {
public int countSubstrings(String s) {
        int n = s.length();

        int[][] a = new int[n][n];
        int count = 0;
        for (int i = 0; i < n; i++) {
            a[i][i] = 1;
            count++;
        }


        for (int col = 1; col < n; col++) {
            for (int row = 0; row < col; row++) {
                if (row == col - 1 && s.charAt(col) == s.charAt(row)) {
                    a[row][col] = 1;
                    count++;
                } else if (a[row + 1][col - 1] == 1 && s.charAt(col) == s.charAt(row) ) {
                     a[row][col] = 1;
                    count++;
                }
            }
        }

        return count;
    }
}
